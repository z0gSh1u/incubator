## 问题

如果让你从零设计一个 AI Agent 的工作流可视化编辑器，你会如何选型和设计架构？

## 回答

### 一、需求分析

AI Agent 工作流编辑器需要支持：

1. **可视化编排**：拖拽式创建节点和连线
2. **多种节点类型**：LLM 调用、API 请求、条件分支、循环等
3. **数据流管理**：节点间的数据传递和转换
4. **调试运行**：单步执行、断点调试、状态回显
5. **版本管理**：保存、导入导出、版本回溯

### 二、技术选型

#### 1. 图形引擎对比

| 库             | 优点                         | 缺点               | 适用场景       |
| -------------- | ---------------------------- | ------------------ | -------------- |
| **React Flow** | React 生态友好、自定义能力强 | 复杂图形性能一般   | 中小规模工作流 |
| **AntV X6**    | 功能丰富、性能优秀           | 学习曲线陡峭       | 大规模复杂图   |
| **JointJS**    | 老牌稳定、文档完善           | 商业版才有高级功能 | 企业级应用     |
| **Rete.js**    | 专为节点编辑器设计           | 社区较小           | 纯节点编辑场景 |

**推荐：React Flow** —— 对 React 开发者友好，社区活跃，自定义能力满足大多数需求。

### 四、状态管理实现

```typescript
// stores/workflowStore.ts
import { create } from 'zustand'
import { temporal } from 'zundo'
import { immer } from 'zustand/middleware/immer'
import {
  Node,
  Edge,
  addEdge,
  applyNodeChanges,
  applyEdgeChanges,
  Connection,
} from 'reactflow'

interface WorkflowState {
  // 画布数据
  nodes: Node[]
  edges: Edge[]

  // 元信息
  workflowId: string | null
  workflowName: string

  // 运行时状态
  isRunning: boolean
  currentNodeId: string | null
  executionLog: ExecutionLogEntry[]

  // 操作方法
  setNodes: (nodes: Node[]) => void
  setEdges: (edges: Edge[]) => void
  onNodesChange: (changes: any[]) => void
  onEdgesChange: (changes: any[]) => void
  onConnect: (connection: Connection) => void

  addNode: (node: Omit<Node, 'id'>) => void
  updateNodeData: (nodeId: string, data: Partial<any>) => void
  deleteNode: (nodeId: string) => void

  // 执行相关
  startExecution: () => void
  stopExecution: () => void
  setNodeStatus: (nodeId: string, status: NodeStatus) => void

  // 持久化
  saveWorkflow: () => Promise<void>
  loadWorkflow: (id: string) => Promise<void>
  exportWorkflow: () => Workflow
  importWorkflow: (workflow: Workflow) => void
}

export const useWorkflowStore = create<WorkflowState>()(
  // temporal 中间件提供撤销重做
  temporal(
    immer((set, get) => ({
      nodes: [],
      edges: [],
      workflowId: null,
      workflowName: 'Untitled Workflow',
      isRunning: false,
      currentNodeId: null,
      executionLog: [],

      setNodes: (nodes) => set({ nodes }),
      setEdges: (edges) => set({ edges }),

      onNodesChange: (changes) =>
        set((state) => {
          state.nodes = applyNodeChanges(changes, state.nodes)
        }),

      onEdgesChange: (changes) =>
        set((state) => {
          state.edges = applyEdgeChanges(changes, state.edges)
        }),

      onConnect: (connection) =>
        set((state) => {
          // 验证连接有效性
          if (validateConnection(connection, state.nodes, state.edges)) {
            state.edges = addEdge(connection, state.edges)
          }
        }),

      addNode: (node) =>
        set((state) => {
          const newNode: Node = {
            ...node,
            id: `node-${Date.now()}-${Math.random().toString(36).substr(2, 9)}`,
          }
          state.nodes.push(newNode)
        }),

      updateNodeData: (nodeId, data) =>
        set((state) => {
          const node = state.nodes.find((n) => n.id === nodeId)
          if (node) {
            node.data = { ...node.data, ...data }
          }
        }),

      deleteNode: (nodeId) =>
        set((state) => {
          state.nodes = state.nodes.filter((n) => n.id !== nodeId)
          state.edges = state.edges.filter(
            (e) => e.source !== nodeId && e.target !== nodeId,
          )
        }),

      startExecution: () =>
        set({
          isRunning: true,
          executionLog: [],
          // 重置所有节点状态
          nodes: get().nodes.map((n) => ({
            ...n,
            data: { ...n.data, status: 'idle' },
          })),
        }),

      stopExecution: () =>
        set({
          isRunning: false,
          currentNodeId: null,
        }),

      setNodeStatus: (nodeId, status) =>
        set((state) => {
          const node = state.nodes.find((n) => n.id === nodeId)
          if (node) {
            node.data.status = status.status
            node.data.result = status.result
            node.data.error = status.error
          }
          state.currentNodeId = nodeId
        }),

      exportWorkflow: () => {
        const { nodes, edges, workflowId, workflowName } = get()
        return {
          id: workflowId || crypto.randomUUID(),
          name: workflowName,
          nodes: nodes.map((n) => ({
            id: n.id,
            type: n.type,
            position: n.position,
            data: n.data,
          })),
          edges: edges.map((e) => ({
            id: e.id,
            source: e.source,
            sourceHandle: e.sourceHandle,
            target: e.target,
            targetHandle: e.targetHandle,
          })),
          variables: {},
          version: '1.0.0',
          createdAt: new Date().toISOString(),
          updatedAt: new Date().toISOString(),
        }
      },

      importWorkflow: (workflow) =>
        set({
          workflowId: workflow.id,
          workflowName: workflow.name,
          nodes: workflow.nodes.map((n) => ({
            ...n,
            type: n.type,
          })),
          edges: workflow.edges,
        }),
    })),
  ),
)

// 验证连接是否有效
function validateConnection(
  connection: Connection,
  nodes: Node[],
  edges: Edge[],
): boolean {
  // 1. 不能连接到自己
  if (connection.source === connection.target) {
    return false
  }

  // 2. 检查是否会形成循环
  if (wouldCreateCycle(connection, nodes, edges)) {
    console.warn('检测到循环引用，拒绝连接')
    return false
  }

  // 3. 检查端口类型是否兼容
  const sourceNode = nodes.find((n) => n.id === connection.source)
  const targetNode = nodes.find((n) => n.id === connection.target)

  if (!sourceNode || !targetNode) return false

  // 可以添加更多类型检查...

  return true
}

// 检测是否会形成循环
function wouldCreateCycle(
  newConnection: Connection,
  nodes: Node[],
  edges: Edge[],
): boolean {
  const adjacencyList = new Map<string, string[]>()

  // 构建邻接表
  for (const edge of edges) {
    if (!adjacencyList.has(edge.source)) {
      adjacencyList.set(edge.source, [])
    }
    adjacencyList.get(edge.source)!.push(edge.target)
  }

  // 添加新连接
  if (!adjacencyList.has(newConnection.source!)) {
    adjacencyList.set(newConnection.source!, [])
  }
  adjacencyList.get(newConnection.source!)!.push(newConnection.target!)

  // DFS 检测循环
  const visited = new Set<string>()
  const recursionStack = new Set<string>()

  function hasCycle(nodeId: string): boolean {
    visited.add(nodeId)
    recursionStack.add(nodeId)

    const neighbors = adjacencyList.get(nodeId) || []
    for (const neighbor of neighbors) {
      if (!visited.has(neighbor)) {
        if (hasCycle(neighbor)) return true
      } else if (recursionStack.has(neighbor)) {
        return true
      }
    }

    recursionStack.delete(nodeId)
    return false
  }

  for (const node of nodes) {
    if (!visited.has(node.id)) {
      if (hasCycle(node.id)) return true
    }
  }

  return false
}
```

### 六、工作流执行引擎

```typescript
// services/workflowExecutor.ts

interface ExecutionContext {
  variables: Record<string, any>
  nodeResults: Map<string, any>
  logs: ExecutionLogEntry[]
}

interface ExecutionLogEntry {
  timestamp: number
  nodeId: string
  type: 'start' | 'end' | 'error' | 'log'
  message: string
  data?: any
}

export class WorkflowExecutor {
  private workflow: Workflow
  private context: ExecutionContext
  private abortController: AbortController
  private onNodeStatusChange: (nodeId: string, status: any) => void

  constructor(
    workflow: Workflow,
    onNodeStatusChange: (nodeId: string, status: any) => void,
  ) {
    this.workflow = workflow
    this.onNodeStatusChange = onNodeStatusChange
    this.abortController = new AbortController()
    this.context = {
      variables: { ...workflow.variables },
      nodeResults: new Map(),
      logs: [],
    }
  }

  async execute(): Promise<ExecutionContext> {
    // 拓扑排序获取执行顺序
    const executionOrder = this.topologicalSort()

    for (const nodeId of executionOrder) {
      if (this.abortController.signal.aborted) {
        throw new Error('Execution aborted')
      }

      await this.executeNode(nodeId)
    }

    return this.context
  }

  private async executeNode(nodeId: string): Promise<void> {
    const node = this.workflow.nodes.find((n) => n.id === nodeId)
    if (!node) throw new Error(`Node ${nodeId} not found`)

    this.log(nodeId, 'start', `开始执行节点: ${node.data.label}`)
    this.onNodeStatusChange(nodeId, { status: 'running' })

    try {
      // 收集输入数据
      const inputs = this.collectInputs(nodeId)

      // 根据节点类型执行
      const result = await this.executeNodeByType(node, inputs)

      // 保存结果
      this.context.nodeResults.set(nodeId, result)

      this.log(nodeId, 'end', `节点执行完成`, result)
      this.onNodeStatusChange(nodeId, { status: 'success', result })
    } catch (error) {
      this.log(nodeId, 'error', `节点执行失败: ${error.message}`)
      this.onNodeStatusChange(nodeId, {
        status: 'error',
        error: error.message,
      })
      throw error
    }
  }

  private async executeNodeByType(
    node: WorkflowNode,
    inputs: Record<string, any>,
  ): Promise<any> {
    switch (node.type) {
      case 'llm':
        return this.executeLLMNode(node, inputs)
      case 'api':
        return this.executeAPINode(node, inputs)
      case 'condition':
        return this.executeConditionNode(node, inputs)
      case 'transform':
        return this.executeTransformNode(node, inputs)
      case 'code':
        return this.executeCodeNode(node, inputs)
      default:
        throw new Error(`Unknown node type: ${node.type}`)
    }
  }

  private async executeLLMNode(
    node: WorkflowNode,
    inputs: Record<string, any>,
  ): Promise<any> {
    const config = node.data.config as LLMNodeConfig

    // 填充 prompt 模板
    const userPrompt = this.fillTemplate(config.userPromptTemplate, {
      ...this.context.variables,
      ...inputs,
    })

    const response = await fetch('/api/llm/chat', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        model: config.model,
        messages: [
          { role: 'system', content: config.systemPrompt },
          { role: 'user', content: userPrompt },
        ],
        temperature: config.temperature,
        max_tokens: config.maxTokens,
      }),
      signal: this.abortController.signal,
    })

    return response.json()
  }

  private async executeConditionNode(
    node: WorkflowNode,
    inputs: Record<string, any>,
  ): Promise<{ branch: string }> {
    const config = node.data.config as ConditionNodeConfig

    // 评估条件
    for (const condition of config.conditions) {
      const result = this.evaluateExpression(condition.expression, {
        ...this.context.variables,
        ...inputs,
      })

      if (result) {
        return { branch: condition.outputHandle }
      }
    }

    return { branch: config.defaultHandle }
  }

  private evaluateExpression(
    expression: string,
    context: Record<string, any>,
  ): boolean {
    // 安全地评估表达式
    const func = new Function(...Object.keys(context), `return ${expression}`)
    return func(...Object.values(context))
  }

  private topologicalSort(): string[] {
    // 构建入度表和邻接表
    const inDegree = new Map<string, number>()
    const adjacencyList = new Map<string, string[]>()

    for (const node of this.workflow.nodes) {
      inDegree.set(node.id, 0)
      adjacencyList.set(node.id, [])
    }

    for (const edge of this.workflow.edges) {
      const targets = adjacencyList.get(edge.source) || []
      targets.push(edge.target)
      adjacencyList.set(edge.source, targets)
      inDegree.set(edge.target, (inDegree.get(edge.target) || 0) + 1)
    }

    // Kahn 算法
    const queue: string[] = []
    const result: string[] = []

    for (const [nodeId, degree] of inDegree) {
      if (degree === 0) {
        queue.push(nodeId)
      }
    }

    while (queue.length > 0) {
      const nodeId = queue.shift()!
      result.push(nodeId)

      for (const neighbor of adjacencyList.get(nodeId) || []) {
        const newDegree = (inDegree.get(neighbor) || 0) - 1
        inDegree.set(neighbor, newDegree)

        if (newDegree === 0) {
          queue.push(neighbor)
        }
      }
    }

    if (result.length !== this.workflow.nodes.length) {
      throw new Error('工作流中存在循环依赖')
    }

    return result
  }

  abort(): void {
    this.abortController.abort()
  }

  private log(
    nodeId: string,
    type: ExecutionLogEntry['type'],
    message: string,
    data?: any,
  ): void {
    this.context.logs.push({
      timestamp: Date.now(),
      nodeId,
      type,
      message,
      data,
    })
  }
}
```

### 九、总结

| 模块         | 技术选型                 | 关键实现             |
| ------------ | ------------------------ | -------------------- |
| **图形引擎** | React Flow               | 节点自定义、事件处理 |
| **状态管理** | Zustand + Immer          | 撤销重做、持久化     |
| **数据模型** | TypeScript + JSON Schema | DAG 结构、类型安全   |
| **执行引擎** | 拓扑排序 + 异步执行      | 支持中断、状态回显   |
| **节点扩展** | 插件化设计               | BaseNode + 具体实现  |

架构设计原则：

1. **数据与视图分离**：工作流数据独立于 React Flow 状态
2. **插件化扩展**：新节点类型只需实现标准接口
3. **安全执行**：表达式沙箱、循环检测、超时保护
4. **性能优先**：虚拟化、增量渲染、WebGL 备选
