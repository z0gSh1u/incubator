## 问题

如何处理 AI 幻觉（Hallucination）在 UI 层的反馈设计？

## 回答

### 一、什么是 AI 幻觉

AI 幻觉指 LLM 生成看起来合理但实际上错误或虚构的内容：

1. **事实性错误**：编造不存在的事实、数据、引用
2. **逻辑矛盾**：前后矛盾的陈述
3. **过度自信**：对不确定内容表现得非常确定
4. **虚假引用**：生成不存在的论文、书籍、链接

前端作为用户与 AI 交互的界面，需要设计合理的机制帮助用户识别和验证 AI 输出。

```tsx
function AnnotatedSegment({
  segment,
  onVerify,
}: {
  segment: ContentSegment
  onVerify?: (segment: ContentSegment) => void
}) {
  const { text, type, confidence, source } = segment

  // 根据类型确定样式

    switch (type) {
      case 'fact':
        return confidence && confidence < 0.7 ? 'segment-uncertain' : 'segment-fact'
      case 'uncertain':
        return 'segment-uncertain'
      case 'citation':
        return 'segment-citation'
      default:
        return 'segment-default'
    }
  }

  return (
    <span className={`content-segment ${getSegmentStyle()}`}>
      {text}

      {/* 低置信度内容显示警告 */}
      {type === 'uncertain' || (confidence && confidence < 0.5) ? (
        <span className="uncertainty-marker" title="此内容可能不准确">
          <ConfidenceIndicator level="low" score={confidence} showLabel={false} />
        </span>
      ) : null}

      {/* 引用来源 */}
      {type === 'citation' && source && (

          <a href={source} target="_blank" rel="noopener">

          </a>
        </sup>
      )}
```

#### 1. 搜索验证组件

```tsx
// components/FactChecker.tsx
import React, { useState, useCallback } from 'react'

interface FactCheckResult {
  query: string
  sources: Array<{
    title: string
    url: string
    snippet: string
    agreement: 'supports' | 'contradicts' | 'neutral'
  }>
  verdict: 'verified' | 'disputed' | 'inconclusive'
  summary: string
}

interface FactCheckerProps {
  statement: string
  onResult?: (result: FactCheckResult) => void
}

export function FactChecker({ statement, onResult }: FactCheckerProps) {
  const [isChecking, setIsChecking] = useState(false)
  const [result, setResult] = useState<FactCheckResult | null>(null)
  const [error, setError] = useState<string | null>(null)

  const handleCheck = useCallback(async () => {
    setIsChecking(true)
    setError(null)

    try {
      // 调用搜索 API 验证
      const response = await fetch('/api/fact-check', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ statement }),
      })

      if (!response.ok) {
        throw new Error('验证服务暂时不可用')
      }

      const checkResult: FactCheckResult = await response.json()
      setResult(checkResult)
      onResult?.(checkResult)
    } catch (err) {
      setError((err as Error).message)
    } finally {
      setIsChecking(false)
    }
  }, [statement, onResult])

  // 一键搜索（备用方案）
  const handleWebSearch = useCallback(() => {
    const query = encodeURIComponent(statement)
    window.open(`https://www.google.com/search?q=${query}`, '_blank')
  }, [statement])

  return (
    <div className="fact-checker">
      <div className="fact-checker-header">
        <span className="statement-preview">"{statement.slice(0, 100)}..."</span>

        <div className="actions">
          <button onClick={handleCheck} disabled={isChecking} className="check-btn">
            {isChecking ? '验证中...' : '🔍 AI 验证'}
          </button>

          <button onClick={handleWebSearch} className="search-btn">
            🌐 网页搜索
          </button>
        </div>
      </div>

      {error && <div className="fact-check-error">⚠️ {error}</div>}

      {result && <FactCheckResult result={result} />}
    </div>
  )
}

function FactCheckResult({ result }: { result: FactCheckResult }) {
  const verdictConfig = {
    verified: { icon: '✅', label: '可验证', color: '#22c55e' },
    disputed: { icon: '❌', label: '有争议', color: '#ef4444' },
    inconclusive: { icon: '❓', label: '无法确认', color: '#f59e0b' },
  }

  const { icon, label, color } = verdictConfig[result.verdict]

  return (
    <div className="fact-check-result">
      <div className="verdict" style={{ borderColor: color, color }}>
        <span className="verdict-icon">{icon}</span>
        <span className="verdict-label">{label}</span>
      </div>

      <p className="summary">{result.summary}</p>

      <div className="sources">
        <h4>参考来源</h4>
        {result.sources.map((source, index) => (
          <div key={index} className={`source-item ${source.agreement}`}>
            <a href={source.url} target="_blank" rel="noopener">
              {source.title}
            </a>
            <p className="snippet">{source.snippet}</p>
            <span className={`agreement-badge ${source.agreement}`}>
              {source.agreement === 'supports' && '✓ 支持'}
              {source.agreement === 'contradicts' && '✗ 矛盾'}
              {source.agreement === 'neutral' && '- 中立'}
            </span>
          </div>
        ))}
      </div>
    </div>
  )
}
```

```tsx
// components/SourceCitation.tsx
import React, { useState, memo } from 'react'

interface Source {
  id: string
  title: string
  content: string
  url?: string
  type: 'document' | 'webpage' | 'database'
  relevanceScore: number
  highlightedText?: string
}

interface SourceCitationProps {
  sources: Source[]
  aiResponse: string
  onSourceClick?: (source: Source) => void
}

export const SourceCitation = memo(function SourceCitation({
  sources,
  aiResponse,
  onSourceClick,
}: SourceCitationProps) {
  const [selectedSource, setSelectedSource] = useState<Source | null>(null)
  const [viewMode, setViewMode] = useState<'inline' | 'sidebar'>('inline')

  return (
    <div className={`source-citation mode-${viewMode}`}>
      {/* AI 响应区域 */}
      <div className="response-section">
        <div className="response-content">
          <ResponseWithCitations
            response={aiResponse}
            sources={sources}
            onCitationClick={(sourceId) => {
              const source = sources.find((s) => s.id === sourceId)
              if (source) {
                setSelectedSource(source)
                onSourceClick?.(source)
              }
            }}
          />
        </div>

        {/* 引用来源列表 */}
        <div className="sources-list">
          <h4>
            📚 引用来源 ({sources.length})
            <button
              className="view-toggle"
              onClick={() =>
                setViewMode((v) => (v === 'inline' ? 'sidebar' : 'inline'))
              }
            >
              {viewMode === 'inline' ? '侧边栏' : '内联'}
            </button>
          </h4>

          {sources
            .sort((a, b) => b.relevanceScore - a.relevanceScore)
            .map((source) => (
              <SourceCard
                key={source.id}
                source={source}
                isSelected={selectedSource?.id === source.id}
                onClick={() => {
                  setSelectedSource(source)
                  onSourceClick?.(source)
                }}
              />
            ))}
        </div>
      </div>

      {/* 对比视图 */}
      {selectedSource && viewMode === 'sidebar' && (
        <div className="comparison-sidebar">
          <div className="sidebar-header">
            <h4>原文对比</h4>
            <button onClick={() => setSelectedSource(null)}>✕</button>
          </div>

          <div className="original-content">
            <span className="source-type">{selectedSource.type}</span>
            <h5>{selectedSource.title}</h5>
            <p className="original-text">
              {selectedSource.highlightedText || selectedSource.content}
            </p>
            {selectedSource.url && (
              <a
                href={selectedSource.url}
                target="_blank"
                rel="noopener"
                className="view-original"
              >
                查看原文 →
              </a>
            )}
          </div>
        </div>
      )}
    </div>
  )
})

// 带引用标记的响应
function ResponseWithCitations({
  response,
  sources,
  onCitationClick,
}: {
  response: string
  sources: Source[]
  onCitationClick: (sourceId: string) => void
}) {
  // 解析响应中的引用标记 [1], [2] 等
  const parsedContent = useMemo(() => {
    const citationRegex = /\[(\d+)\]/g
    const parts: Array<{ type: 'text' | 'citation'; content: string }> = []
    let lastIndex = 0
    let match

    while ((match = citationRegex.exec(response)) !== null) {
      // 添加引用前的文本
      if (match.index > lastIndex) {
        parts.push({
          type: 'text',
          content: response.slice(lastIndex, match.index),
        })
      }

      // 添加引用
      parts.push({
        type: 'citation',
        content: match[1],
      })

      lastIndex = match.index + match[0].length
    }

    // 添加剩余文本
    if (lastIndex < response.length) {
      parts.push({
        type: 'text',
        content: response.slice(lastIndex),
      })
    }

    return parts
  }, [response])

  return (
    <div className="response-with-citations">
      {parsedContent.map((part, index) => {
        if (part.type === 'text') {
          return <span key={index}>{part.content}</span>
        }

        const sourceIndex = parseInt(part.content) - 1
        const source = sources[sourceIndex]

        return (
          <sup
            key={index}
            className="citation-marker"
            onClick={() => source && onCitationClick(source.id)}
            title={source?.title || '来源'}
          >
            [{part.content}]
          </sup>
        )
      })}
    </div>
  )
}

function SourceCard({
  source,
  isSelected,
  onClick,
}: {
  source: Source
  isSelected: boolean
  onClick: () => void
}) {
  const typeIcons = {
    document: '📄',
    webpage: '🌐',
    database: '🗄️',
  }

  return (
    <div className={`source-card ${isSelected ? 'selected' : ''}`} onClick={onClick}>
      <span className="source-icon">{typeIcons[source.type]}</span>
      <div className="source-info">
        <span className="source-title">{source.title}</span>
        <span className="relevance-score">
          相关度: {(source.relevanceScore * 100).toFixed(0)}%
        </span>
      </div>
    </div>
  )
}
```

```tsx
// components/CopyWithWarning.tsx
import React, { useCallback, useState } from 'react'

interface CopyWithWarningProps {
  content: string
  showWarning?: boolean
  warningMessage?: string
  onCopy?: () => void
}

export function CopyWithWarning({
  content,
  showWarning = true,
  warningMessage = '请注意：AI 生成内容可能存在错误，请核实后使用',
  onCopy,
}: CopyWithWarningProps) {
  const [copied, setCopied] = useState(false)
  const [showModal, setShowModal] = useState(false)

  const handleCopy = useCallback(async () => {
    if (showWarning) {
      setShowModal(true)
    } else {
      await performCopy()
    }
  }, [showWarning])

  const performCopy = async () => {
    try {
      await navigator.clipboard.writeText(content)
      setCopied(true)
      onCopy?.()
      setTimeout(() => setCopied(false), 2000)
    } catch (err) {
      console.error('复制失败:', err)
    }
  }

  const handleConfirm = async () => {
    setShowModal(false)
    await performCopy()
  }

  return (
    <>
      <button className={`copy-btn ${copied ? 'copied' : ''}`} onClick={handleCopy}>
        {copied ? '✓ 已复制' : '📋 复制'}
      </button>

      {showModal && (
        <div className="warning-modal-overlay">
          <div className="warning-modal">
            <div className="warning-icon">⚠️</div>
            <p className="warning-message">{warningMessage}</p>

            <div className="warning-actions">
              <button className="cancel-btn" onClick={() => setShowModal(false)}>
                取消
              </button>
              <button className="confirm-btn" onClick={handleConfirm}>
                我已知晓，继续复制
              </button>
            </div>

            <label className="dont-show-again">
              <input type="checkbox" />
              <span>不再提示</span>
            </label>
          </div>
        </div>
      )}
    </>
  )
}
```

### 六、用户反馈收集

```tsx
// components/HallucinationFeedback.tsx
import React, { useState, useCallback } from 'react'

interface FeedbackData {
  messageId: string
  feedbackType: 'accurate' | 'inaccurate' | 'uncertain'
  issueTypes?: string[]
  correction?: string
  additionalInfo?: string
}

interface HallucinationFeedbackProps {
  messageId: string
  content: string
  onSubmit: (feedback: FeedbackData) => void
}

export function HallucinationFeedback({
  messageId,
  content,
  onSubmit,
}: HallucinationFeedbackProps) {
  const [feedbackType, setFeedbackType] = useState<FeedbackData['feedbackType'] | null>(
    null,
  )
  const [issueTypes, setIssueTypes] = useState<string[]>([])
  const [correction, setCorrection] = useState('')
  const [submitted, setSubmitted] = useState(false)

  const issueOptions = [
    { id: 'factual_error', label: '事实错误' },
    { id: 'outdated_info', label: '信息过时' },
    { id: 'fabricated_source', label: '虚假引用' },
    { id: 'logical_inconsistency', label: '逻辑矛盾' },
    { id: 'misleading', label: '误导性内容' },
    { id: 'incomplete', label: '信息不完整' },
  ]

  const handleSubmit = useCallback(() => {
    if (!feedbackType) return

    const feedback: FeedbackData = {
      messageId,
      feedbackType,
      issueTypes: feedbackType === 'inaccurate' ? issueTypes : undefined,
      correction: correction || undefined,
    }

    onSubmit(feedback)
    setSubmitted(true)
  }, [messageId, feedbackType, issueTypes, correction, onSubmit])

  if (submitted) {
    return (
      <div className="feedback-submitted">
        ✓ 感谢您的反馈，这将帮助我们改进 AI 输出质量
      </div>
    )
  }

  return (
    <div className="hallucination-feedback">
      <div className="feedback-prompt">这个回答准确吗？</div>

      <div className="feedback-buttons">
        <button
          className={`feedback-btn ${feedbackType === 'accurate' ? 'selected' : ''}`}
          onClick={() => setFeedbackType('accurate')}
        >
          👍 准确
        </button>
        <button
          className={`feedback-btn ${feedbackType === 'inaccurate' ? 'selected' : ''}`}
          onClick={() => setFeedbackType('inaccurate')}
        >
          👎 不准确
        </button>
        <button
          className={`feedback-btn ${feedbackType === 'uncertain' ? 'selected' : ''}`}
          onClick={() => setFeedbackType('uncertain')}
        >
          🤔 不确定
        </button>
      </div>

      {feedbackType === 'inaccurate' && (
        <div className="feedback-details">
          <div className="issue-types">
            <p>请选择问题类型（可多选）：</p>
            {issueOptions.map((option) => (
              <label key={option.id} className="issue-option">
                <input
                  type="checkbox"
                  checked={issueTypes.includes(option.id)}
                  onChange={(e) => {
                    if (e.target.checked) {
                      setIssueTypes([...issueTypes, option.id])
                    } else {
                      setIssueTypes(issueTypes.filter((t) => t !== option.id))
                    }
                  }}
                />
                <span>{option.label}</span>
              </label>
            ))}
          </div>

          <div className="correction-input">
            <label>提供正确信息（可选）：</label>
            <textarea
              value={correction}
              onChange={(e) => setCorrection(e.target.value)}
              placeholder="请输入正确的信息..."
              rows={3}
            />
          </div>
        </div>
      )}

      {feedbackType && (
        <button className="submit-feedback-btn" onClick={handleSubmit}>
          提交反馈
        </button>
      )}
    </div>
  )
}
```

### 七、完整消息组件整合

```tsx
// components/AIMessage.tsx
import React, { memo, useState } from 'react'
import { ConfidenceIndicator } from './ConfidenceIndicator'
import { SourceCitation } from './SourceCitation'
import { FactChecker } from './FactChecker'
import { CopyWithWarning } from './CopyWithWarning'
import { AIDisclaimer } from './AIDisclaimer'
import { HallucinationFeedback } from './HallucinationFeedback'

interface AIMessageProps {
  message: {
    id: string
    content: string
    confidence?: number
    sources?: Source[]
    model?: string
  }
  showFactCheck?: boolean
}

export const AIMessage = memo(function AIMessage({
  message,
  showFactCheck = true,
}: AIMessageProps) {
  const [selectedText, setSelectedText] = useState<string | null>(null)
  const { id, content, confidence, sources, model } = message

  // 计算置信度级别
  const confidenceLevel = confidence
    ? confidence > 0.8
      ? 'high'
      : confidence > 0.5
        ? 'medium'
        : 'low'
    : 'uncertain'

  // 处理文本选择
  const handleTextSelection = () => {
    const selection = window.getSelection()
    if (selection && selection.toString().trim().length > 10) {
      setSelectedText(selection.toString())
    }
  }

  return (
    <div className="ai-message">
      {/* 消息头部 */}
      <div className="message-header">
        <span className="ai-avatar">🤖</span>
        <span className="ai-label">AI 助手</span>
        {model && <span className="model-badge">{model}</span>}
        <ConfidenceIndicator level={confidenceLevel} score={confidence} />
      </div>

      {/* 消息内容 */}
      <div className="message-content" onMouseUp={handleTextSelection}>
        {sources && sources.length > 0 ? (
          <SourceCitation sources={sources} aiResponse={content} />
        ) : (
          <div className="plain-content">{content}</div>
        )}
      </div>

      {/* 选中文本的事实检查 */}
      {selectedText && showFactCheck && (
        <div className="inline-fact-check">
          <FactChecker
            statement={selectedText}
            onResult={() => setSelectedText(null)}
          />
        </div>
      )}

      {/* 操作栏 */}
      <div className="message-actions">
        <CopyWithWarning content={content} showWarning={confidenceLevel !== 'high'} />
        <button className="regenerate-btn" title="重新生成">
          🔄
        </button>
      </div>

      {/* 免责声明 */}
      <AIDisclaimer variant="inline" model={model} />

      {/* 反馈收集 */}
      <HallucinationFeedback
        messageId={id}
        content={content}
        onSubmit={(feedback) => {
          console.log('用户反馈:', feedback)
          // 上报到分析服务
        }}
      />
    </div>
  )
})
```

### 八、总结

| 设计策略         | 实现方式            | 用户价值             |
| ---------------- | ------------------- | -------------------- |
| **置信度可视化** | 标记低置信内容      | 识别可能不准确的部分 |
| **事实检查**     | 搜索验证按钮        | 快速验证关键信息     |
| **引用溯源**     | 来源标注 + 对比视图 | 追溯信息来源         |
| **交互引导**     | 复制提醒 + 免责声明 | 提醒用户审慎使用     |
| **反馈收集**     | 结构化反馈表单      | 持续优化模型         |

核心原则：

1. **透明性**：明确告知内容由 AI 生成
2. **可验证**：提供验证工具而非盲目信任
3. **可追溯**：展示知识来源便于核实
4. **可反馈**：收集用户反馈持续改进
