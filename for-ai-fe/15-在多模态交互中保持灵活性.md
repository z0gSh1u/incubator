## 问题

在多模态（图片、语音、视频）交互中，前端架构如何保持灵活性？

## 回答

### 一、多模态交互的挑战

AI 应用正从纯文本扩展到多模态交互：

1. **输入多样化**：文本、图片、语音、视频、文件
2. **输出多样化**：文本、图片生成、语音合成、结构化数据
3. **处理复杂化**：需要前端预处理（压缩、转码）
4. **扩展性要求**：新模态类型随时可能增加

#### 2. 消息序列化与反序列化

```typescript
// utils/messageSerializer.ts

class MessageSerializer {
  /** 序列化为发送格式 */
  static toAPI(message: Message): APIMessage {
    return {
      role: message.role,
      content: message.content.map((block) => {
        switch (block.type) {
          case 'text':
            return { type: 'text', text: block.text }
          case 'image':
            return {
              type: 'image_url',
              image_url: { url: block.url },
            }
          case 'audio':
            return {
              type: 'input_audio',
              input_audio: {
                data: block.url,
                format: 'wav',
              },
            }
          default:
            return { type: 'text', text: `[${block.type}]` }
        }
      }),
    }
  }

  /** 从 API 响应反序列化 */
  static fromAPI(response: any): Message {
    const content: ContentBlock[] = []

    // 处理多种响应格式
    if (typeof response.content === 'string') {
      content.push({
        id: crypto.randomUUID(),
        type: 'text',
        text: response.content,
        createdAt: Date.now(),
      })
    } else if (Array.isArray(response.content)) {
      for (const block of response.content) {
        content.push(this.parseContentBlock(block))
      }
    }

    return {
      id: response.id || crypto.randomUUID(),
      role: response.role || 'assistant',
      content,
      metadata: {
        model: response.model,
        tokens: response.usage?.total_tokens,
      },
      createdAt: Date.now(),
      updatedAt: Date.now(),
    }
  }

  private static parseContentBlock(block: any): ContentBlock {
    // 根据类型解析
    switch (block.type) {
      case 'text':
        return {
          id: crypto.randomUUID(),
          type: 'text',
          text: block.text,
          createdAt: Date.now(),
        }
      case 'image':
        return {
          id: crypto.randomUUID(),
          type: 'image',
          url: block.url || block.image_url?.url,
          mimeType: 'image/png',
          size: 0,
          createdAt: Date.now(),
        }
      default:
        return {
          id: crypto.randomUUID(),
          type: 'text',
          text: JSON.stringify(block),
          createdAt: Date.now(),
        }
    }
  }
}
```

```tsx
// renderers/RendererRegistry.tsx
import React, { ComponentType, lazy, Suspense } from 'react'

/** 渲染器 Props */
interface RendererProps<T extends ContentBlock = ContentBlock> {
  content: T
  isStreaming?: boolean
  onAction?: (action: string, payload?: any) => void
}

/** 渲染器配置 */
interface RendererConfig {
  component: ComponentType<RendererProps<any>>
  // 是否支持流式渲染
  supportsStreaming?: boolean
  // 懒加载
  lazy?: boolean
}

class RendererRegistry {
  private static renderers = new Map<ModalityType, RendererConfig>()
  private static fallback: ComponentType<RendererProps> = DefaultRenderer

  /** 注册渲染器 */
  static register(type: ModalityType, config: RendererConfig): void {
    this.renderers.set(type, config)
  }

  /** 批量注册 */
  static registerAll(configs: Record<ModalityType, RendererConfig>): void {
    for (const [type, config] of Object.entries(configs)) {
      this.register(type as ModalityType, config)
    }
  }

  /** 获取渲染器 */
  static get(type: ModalityType): ComponentType<RendererProps> {
    const config = this.renderers.get(type)

    if (!config) {
      console.warn(`No renderer for type: ${type}, using fallback`)
      return this.fallback
    }

    return config.component
  }

  /** 检查是否支持流式 */
  static supportsStreaming(type: ModalityType): boolean {
    return this.renderers.get(type)?.supportsStreaming ?? false
  }

  /** 设置回退渲染器 */
  static setFallback(component: ComponentType<RendererProps>): void {
    this.fallback = component
  }
}

// 默认渲染器
function DefaultRenderer({ content }: RendererProps) {
  return (
    <div className="unknown-content">
      <span className="type-badge">{content.type}</span>
      <pre>{JSON.stringify(content, null, 2)}</pre>
    </div>
  )
}

export { RendererRegistry, RendererProps }
```

```typescript
// inputs/InputPluginManager.ts

/** 输入插件接口 */
interface InputPlugin {
  id: string
  name: string
  icon: string

  // 插件是否可用
  isAvailable: () => boolean | Promise<boolean>

  // 激活插件（如打开文件选择器）
  activate: () => Promise<InputResult | null>

  // 渲染插件 UI（如录音按钮）
  renderTrigger?: () => React.ReactNode

  // 渲染输入预览
  renderPreview?: (data: any) => React.ReactNode
}

/** 输入结果 */
interface InputResult {
  type: ModalityType
  data: Blob | string | File
  metadata?: Record<string, any>
  preview?: string
}

class InputPluginManager {
  private plugins: Map<string, InputPlugin> = new Map()

  /** 注册插件 */
  register(plugin: InputPlugin): void {
    this.plugins.set(plugin.id, plugin)
  }

  /** 获取所有可用插件 */
  async getAvailablePlugins(): Promise<InputPlugin[]> {
    const available: InputPlugin[] = []

    for (const plugin of this.plugins.values()) {
      if (await plugin.isAvailable()) {
        available.push(plugin)
      }
    }

    return available
  }

  /** 激活指定插件 */
  async activatePlugin(id: string): Promise<InputResult | null> {
    const plugin = this.plugins.get(id)
    if (!plugin) {
      throw new Error(`Plugin ${id} not found`)
    }
    return plugin.activate()
  }

  /** 获取插件 */
  getPlugin(id: string): InputPlugin | undefined {
    return this.plugins.get(id)
  }
}

export const inputPluginManager = new InputPluginManager()
```

```tsx
            originalSize: file.size,
            compressedSize: compressed.size,
            mimeType: compressed.type,

        })

// 图片压缩工具
async function compressImage(
  file: File,
  options: { maxWidth: number; maxHeight: number; quality: number },
): Promise<Blob> {
  return new Promise((resolve, reject) => {
    const img = new Image()
    img.onload = () => {
      const canvas = document.createElement('canvas')
      let { width, height } = img

      // 计算缩放比例
      if (width > options.maxWidth) {
        height = (height * options.maxWidth) / width
        width = options.maxWidth
      }
      if (height > options.maxHeight) {
        width = (width * options.maxHeight) / height
        height = options.maxHeight
      }

      canvas.width = width
      canvas.height = height

      const ctx = canvas.getContext('2d')!
      ctx.drawImage(img, 0, 0, width, height)

      canvas.toBlob(
        (blob) => {
          if (blob) resolve(blob)
          else reject(new Error('压缩失败'))
        },
        'image/jpeg',
        options.quality,
      )
    }
    img.onerror = reject
    img.src = URL.createObjectURL(file)
  })
}
```

```tsx
export const voiceRecordPlugin: InputPlugin = {
  id: 'voice-record',
  name: '语音',
  icon: '🎤',

  isAvailable: async () => {
    try {
      const devices = await navigator.mediaDevices.enumerateDevices()
      return devices.some((d) => d.kind === 'audioinput')
    } catch {
      return false
    }
  },

  activate: async () => {
    return new Promise((resolve, reject) => {
      // 创建录音对话框

          const wavBlob = await convertToWav(audioBlob)

          resolve({
            type: 'audio',
            data: wavBlob,
            metadata: {
              duration,
              mimeType: 'audio/wav',
              size: wavBlob.size,
            },
          })
        },
        onCancel: () => resolve(null),
        onError: reject,

      recorder.start()
    })

// 语音录制类
class VoiceRecorder {
  private mediaRecorder: MediaRecorder | null = null
  private audioChunks: Blob[] = []
  private startTime: number = 0

  constructor(
    private callbacks: {
      onComplete: (blob: Blob, duration: number) => void
      onCancel: () => void
      onError: (error: Error) => void
    },
  ) {}

  async start(): Promise<void> {
    try {
      const stream = await navigator.mediaDevices.getUserMedia({
        audio: {
          echoCancellation: true,
          noiseSuppression: true,
          sampleRate: 16000,
        },
      })

      this.mediaRecorder = new MediaRecorder(stream, {
        mimeType: 'audio/webm;codecs=opus',
      })

      this.audioChunks = []
      this.startTime = Date.now()

      this.mediaRecorder.ondataavailable = (e) => {
        this.audioChunks.push(e.data)
      }

      this.mediaRecorder.onstop = () => {
        const duration = (Date.now() - this.startTime) / 1000
        const blob = new Blob(this.audioChunks, { type: 'audio/webm' })
        this.callbacks.onComplete(blob, duration)

        // 清理资源
        stream.getTracks().forEach((track) => track.stop())
      }

      this.mediaRecorder.start(100) // 每 100ms 收集一次数据
    } catch (error) {
      this.callbacks.onError(error as Error)
    }
  }

  stop(): void {
    this.mediaRecorder?.stop()
  }

  cancel(): void {
    this.mediaRecorder?.stop()
    this.callbacks.onCancel()
  }
}

// WebAudio API 转换为 WAV
async function convertToWav(blob: Blob): Promise<Blob> {
  const audioContext = new AudioContext({ sampleRate: 16000 })
  const arrayBuffer = await blob.arrayBuffer()
  const audioBuffer = await audioContext.decodeAudioData(arrayBuffer)

  // 转换为 WAV
  const wavData = audioBufferToWav(audioBuffer)
  return new Blob([wavData], { type: 'audio/wav' })
}
```

```tsx
// components/MultimodalInput.tsx
import React, { useState, useRef, useCallback } from 'react'
import { inputPluginManager, InputResult } from '../inputs/InputPluginManager'

interface MultimodalInputProps {
  onSend: (content: ContentBlock[]) => void
  disabled?: boolean
  placeholder?: string
}

export function MultimodalInput({
  onSend,
  disabled,
  placeholder = '输入消息...',
}: MultimodalInputProps) {
  const [text, setText] = useState('')
  const [attachments, setAttachments] = useState<InputResult[]>([])
  const [availablePlugins, setAvailablePlugins] = useState<InputPlugin[]>([])
  const textareaRef = useRef<HTMLTextAreaElement>(null)

  // 加载可用插件
  useEffect(() => {
    inputPluginManager.getAvailablePlugins().then(setAvailablePlugins)
  }, [])

  // 激活插件
  const handlePluginActivate = useCallback(async (pluginId: string) => {
    const result = await inputPluginManager.activatePlugin(pluginId)
    if (result) {
      setAttachments((prev) => [...prev, result])
    }
  }, [])

  // 移除附件
  const handleRemoveAttachment = useCallback((index: number) => {
    setAttachments((prev) => prev.filter((_, i) => i !== index))
  }, [])

  // 发送消息
  const handleSend = useCallback(async () => {
    if (!text.trim() && attachments.length === 0) return

    const content: ContentBlock[] = []

    // 处理附件
    for (const attachment of attachments) {
      const uploaded = await uploadAttachment(attachment)
      content.push(uploaded)
    }

    // 添加文本
    if (text.trim()) {
      content.push({
        id: crypto.randomUUID(),
        type: 'text',
        text: text.trim(),
        createdAt: Date.now(),
      })
    }

    onSend(content)
    setText('')
    setAttachments([])
  }, [text, attachments, onSend])

  // 处理粘贴
  const handlePaste = useCallback(async (e: React.ClipboardEvent) => {
    const items = e.clipboardData.items

    for (const item of items) {
      if (item.type.startsWith('image/')) {
        e.preventDefault()
        const file = item.getAsFile()
        if (file) {
          const result = await processImage(file)
          setAttachments((prev) => [...prev, result])
        }
      }
    }
  }, [])

  return (
    <div className="multimodal-input">
      {/* 附件预览区 */}
      {attachments.length > 0 && (
        <div className="attachments-preview">
          {attachments.map((attachment, index) => {
            const plugin = inputPluginManager.getPlugin(`${attachment.type}-upload`)

            return (
              <div key={index} className="attachment-item">
                {plugin?.renderPreview?.(attachment) || <span>{attachment.type}</span>}
                <button
                  className="remove-btn"
                  onClick={() => handleRemoveAttachment(index)}
                >
                  ×
                </button>
              </div>
            )
          })}
        </div>
      )}

      {/* 输入区域 */}
      <div className="input-area">
        {/* 插件按钮 */}
        <div className="plugin-buttons">
          {availablePlugins.map((plugin) => (
            <button
              key={plugin.id}
              className="plugin-btn"
              onClick={() => handlePluginActivate(plugin.id)}
              title={plugin.name}
            >
              {plugin.renderTrigger?.() || plugin.icon}
            </button>
          ))}
        </div>

        {/* 文本输入 */}
        <textarea
          ref={textareaRef}
          value={text}
          onChange={(e) => setText(e.target.value)}
          onPaste={handlePaste}
          onKeyDown={(e) => {
            if (e.key === 'Enter' && !e.shiftKey) {
              e.preventDefault()
              handleSend()
            }
          }}
          placeholder={placeholder}
          disabled={disabled}
          rows={1}
        />

        {/* 发送按钮 */}
        <button
          className="send-btn"
          onClick={handleSend}
          disabled={disabled || (!text.trim() && attachments.length === 0)}
        >
          发送
        </button>
      </div>
    </div>
  )
}
```

```typescript
// pipeline/PreprocessPipeline.ts

/** 预处理器接口 */
interface Preprocessor<T = any, R = any> {
  name: string
  // 是否应该处理此输入
  shouldProcess: (input: T) => boolean
  // 处理逻辑
  process: (input: T) => Promise<R>
}

class PreprocessPipeline {
  private preprocessors: Preprocessor[] = []

  /** 添加预处理器 */
  use(preprocessor: Preprocessor): this {
    this.preprocessors.push(preprocessor)
    return this
  }

  /** 执行处理 */
  async process(input: any): Promise<any> {
    let result = input

    for (const preprocessor of this.preprocessors) {
      if (preprocessor.shouldProcess(result)) {
        console.log(`[Pipeline] Running: ${preprocessor.name}`)
        result = await preprocessor.process(result)
      }
    }

    return result
  }
}

// 图片预处理器
const imagePreprocessor: Preprocessor<InputResult> = {
  name: 'image-compress',

  shouldProcess: (input) => input.type === 'image',

  async process(input) {
    const compressed = await compressImage(input.data as File, {
      maxWidth: 1920,
      maxHeight: 1080,
      quality: 0.8,
    })

    return {
      ...input,
      data: compressed,
      metadata: {
        ...input.metadata,
        compressed: true,
        compressedSize: compressed.size,
      },
    }
  },
}

// 音频预处理器
const audioPreprocessor: Preprocessor<InputResult> = {
  name: 'audio-transcode',

  shouldProcess: (input) => input.type === 'audio',

  async process(input) {
    // 转换为 16kHz mono WAV（Whisper 要求格式）
    const wavBlob = await transcodeAudio(input.data as Blob, {
      sampleRate: 16000,
      channels: 1,
      format: 'wav',
    })

    return {
      ...input,
      data: wavBlob,
      metadata: {
        ...input.metadata,
        transcoded: true,
        format: 'wav',
        sampleRate: 16000,
      },
    }
  },
}

// 视频预处理器
const videoPreprocessor: Preprocessor<InputResult> = {
  name: 'video-extract-frames',

  shouldProcess: (input) => input.type === 'video',

  async process(input) {
    // 抽取关键帧
    const frames = await extractVideoFrames(input.data as File, {
      interval: 1000, // 每秒抽一帧
      maxFrames: 10,
    })

    return {
      ...input,
      metadata: {
        ...input.metadata,
        frames,
        frameCount: frames.length,
      },
    }
  },
}

// 创建处理管道
export const preprocessPipeline = new PreprocessPipeline()
  .use(imagePreprocessor)
  .use(audioPreprocessor)
  .use(videoPreprocessor)
```

### 六、总结

| 模块         | 设计模式            | 核心能力         |
| ------------ | ------------------- | ---------------- |
| **消息协议** | 统一类型 + 联合类型 | 支持混合模态消息 |
| **渲染工厂** | 注册中心 + 策略模式 | 按类型分发渲染器 |
| **输入插件** | 插件管理器          | 可扩展的输入方式 |
| **预处理**   | 管道模式            | 压缩、转码、抽帧 |

扩展新模态的步骤：

1. 在 `ContentBlock` 联合类型中添加新类型
2. 实现对应的 `Renderer` 并注册
3. 实现对应的 `InputPlugin` 并注册
4. 添加预处理器（如需要）
